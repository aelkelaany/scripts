SELECT DISTINCT
SIRASGN_PIDM AS THE_PIDM,
THE_SIRDPCL.SIRDPCL_COLL_CODE AS "—ﬁ„ «·ﬂ·Ì…",
STVCOLL_DESC AS "«”„ «·ﬂ·Ì…",
THE_SIRDPCL.SIRDPCL_DEPT_CODE AS "—ﬁ„ «·ﬁ”„",
STVDEPT_DESC AS "«”„ «·ﬁ”„",
SPRIDEN_ID AS "—ﬁ„ «·„œ—”",
SPRIDEN_FIRST_NAME||' '||SPRIDEN_MI||' '||SPRIDEN_LAST_NAME AS "«”„ «·„œ—”"
FROM SIRASGN , SPRIDEN , STVDEPT , STVCOLL , 
(
SELECT DISTINCT
SIRDPCL_PIDM,
SIRDPCL_COLL_CODE,
SIRDPCL_DEPT_CODE
FROM SIRDPCL SI
WHERE TRIM(BOTH ' ' FROM SIRDPCL_TERM_CODE_EFF)=(SELECT MAX(TRIM(BOTH ' ' FROM SIRDPCL_TERM_CODE_EFF)) 
                                                 FROM SIRDPCL 
                                                 WHERE SIRDPCL_PIDM=SI.SIRDPCL_PIDM)
) THE_SIRDPCL
WHERE SIRASGN_TERM_CODE=:THE_TERM
AND SIRASGN_PIDM=THE_SIRDPCL.SIRDPCL_PIDM(+)
AND SIRASGN_PIDM=SPRIDEN_PIDM(+)
AND SPRIDEN_CHANGE_IND IS NULL
AND THE_SIRDPCL.SIRDPCL_DEPT_CODE=STVDEPT_CODE(+)
AND THE_SIRDPCL.SIRDPCL_COLL_CODE=STVCOLL_CODE(+)
ORDER BY THE_SIRDPCL.SIRDPCL_COLL_CODE ASC , THE_SIRDPCL.SIRDPCL_DEPT_CODE ASC
