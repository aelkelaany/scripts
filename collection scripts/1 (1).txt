SELECT DISTINCT
SSRMEET_TERM_CODE,
SSRMEET_CRN,
SSBSECT_SUBJ_CODE,
SSBSECT_CRSE_NUMB,
SSRMEET_SCHD_CODE,
SCBCRSE_TITLE,
SSBSECT_CRSE_TITLE,
SSBSECT_CAMP_CODE,
SSRMEET_BLDG_CODE,
SSRMEET_ROOM_CODE,
SCBCRSE_COLL_CODE,
SCBCRSE_DEPT_CODE,
SCBCRSE_CREDIT_HR_LOW,
SCBCRSE_LEC_HR_LOW,
SCBCRSE_LAB_HR_LOW,
SSBSECT_MAX_ENRL,
SSBSECT_ENRL,
SSBSECT_SEATS_AVAIL,
SSRMEET_BEGIN_TIME,
SSRMEET_END_TIME,
SSRMEET_SAT_DAY,
SSRMEET_SUN_DAY,
SSRMEET_MON_DAY,
SSRMEET_TUE_DAY,
SSRMEET_WED_DAY,
SSRMEET_THU_DAY,
SSRMEET_FRI_DAY,
SSRMEET_USER_ID,
SSBSECT_USER_ID
FROM  SSBSECT , SSRMEET ,
(
SELECT DISTINCT
SCBCRSE_SUBJ_CODE,
SCBCRSE_CRSE_NUMB,
SCBCRSE_COLL_CODE,
SCBCRSE_DEPT_CODE,
SCBCRSE_TITLE,
SCBCRSE_CREDIT_HR_LOW,
SCBCRSE_LEC_HR_LOW,
SCBCRSE_LAB_HR_LOW,
SCBCRSE_USER_ID
FROM SCBCRSE SC
WHERE SCBCRSE_EFF_TERM=(SELECT max(SCBCRSE_EFF_TERM)
                        FROM SCBCRSE 
                        WHERE SCBCRSE_CRSE_NUMB=SC.SCBCRSE_CRSE_NUMB
                        AND SCBCRSE_SUBJ_CODE=SC.SCBCRSE_SUBJ_CODE)
) THE_SCBCRSE                       
WHERE SSBSECT_TERM_CODE=:THE_TERM
AND SSBSECT_CRN=SSRMEET_CRN(+)
AND SSBSECT_TERM_CODE=SSRMEET_TERM_CODE(+)
AND SSBSECT_SUBJ_CODE=THE_SCBCRSE.SCBCRSE_SUBJ_CODE(+)
AND SSBSECT_CRSE_NUMB=THE_SCBCRSE.SCBCRSE_CRSE_NUMB(+)
AND SSBSECT_CRN=:THE_CRN
